plt.figure(figsize=(10, 8))

    data=df_sorted,
    x='Value',
    y='Mortality Cause',
    order=order_list, 
    linewidth=0.2,
    color='steelblue')

plt.title('Wolf Mortality Cause, 1995-2022', fontsize=14, pad=20)

            'Source: Cassidy et al., 2024',
