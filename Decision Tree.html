MORTALITY_DATA = {
    'Summer': {
        'CONTROL ACTION': 1.13, 'MALNUTRITION': 1.13, 'UNKNOWN': 1.13,
        'DISEASE': 1.69, 'ILLEGAL': 1.69, 'NATURAL OTHER': 2.82,
        'VEHICLE': 3.95, 'INTERSPECIFIC': 7.34, 'NATURAL UNKNOWN': 9.04,
        'INTRASPECIFIC': 21.47
    },
    'Winter': {
        'DISEASE': 1.13, 'VEHICLE': 1.13, 'INTERSPECIFIC': 1.13,
        'NATURAL OTHER': 1.69, 'MALNUTRITION': 2.26, 'ILLEGAL': 2.82,
        'UNKNOWN': 2.82, 'NATURAL UNKNOWN': 5.08, 'HARVEST': 10.17,
        'INTRASPECIFIC': 20.34
    }
}


NATURAL_CAUSES = {'INTRASPECIFIC', 'NATURAL UNKNOWN', 'MALNUTRITION', 'NATURAL OTHER', 'INTERSPECIFIC', 'DISEASE'}
HUMAN_CAUSES = {'HARVEST', 'ILLEGAL', 'VEHICLE', 'CONTROL ACTION'}
UNKNOWN_CAUSES = {'UNKNOWN'}

def get_primary_cause(cause):
    """Maps the specific cause to the primary cause group."""
    if cause in NATURAL_CAUSES:
        return 'Natural'
    elif cause in HUMAN_CAUSES:
        return 'Human'
    elif cause in UNKNOWN_CAUSES:
        return 'Unknown'
    return 'Other'


data = []
for season, causes in MORTALITY_DATA.items():
    for cause, percent in causes.items():
        primary_cause = get_primary_cause(cause)
        num_rows = int(percent * 100) 
        for _ in range(num_rows):
            data.append([season, primary_cause, cause])

df = pd.DataFrame(data, columns=['Season', 'Primary_Cause', 'Specific_Cause'])
print("--- Synthetic Data Snapshot (First 5 Rows) ---")
print(df.head())
print(f"\nTotal synthetic rows created: {len(df)}")

le_season = LabelEncoder()
le_primary = LabelEncoder()
le_specific = LabelEncoder()


df['Season_Encoded'] = le_season.fit_transform(df['Season'])
df['Primary_Cause_Encoded'] = le_primary.fit_transform(df['Primary_Cause'])


df['Specific_Cause_Encoded'] = le_specific.fit_transform(df['Specific_Cause'])


X = df[['Season_Encoded', 'Primary_Cause_Encoded']]
y = df['Specific_Cause_Encoded']


X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.1, random_state=42)


model = DecisionTreeClassifier(random_state=42, max_depth=3) 
model.fit(X_train, y_train)


accuracy = model.score(X_test, y_test)
print(f"\nModel Accuracy: {accuracy * 100:.2f}%")

Model Accuracy: 59.96%
